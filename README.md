# INSTALL LIB DOTENV

`yarn add dotenv`

# INSTALL AXIOS

`yarn add axios`
`yarn add @types/axios -D`

# INSTALL LIB JSONWEBTOKEN

`yarn add jsonwebtoken`

`yarn add @types/jsonwebtoken -D`

# INSTALL LIB SOCKET IO

`yarn add socket.io`
`yarn add @types/socket.io -D`

# INSTALL LIB CORS

`yarn add cors`
`yarn add @types/cors -D`
